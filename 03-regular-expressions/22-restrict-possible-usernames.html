Created a system to test a series of usernames
const usernames = [
"JackOfAllTrades",
"JACK",
"J",
"Oceans11",
"B34ns",
"RegexGuru",
"007",
"9"
];

let userCheck = /^[a-z]{2,}\d*$/i;

usernames.forEach (function(username){
userCheck.test(username) ? console.log(username + " is VALID") : console.log(username + " is INVALID");
});
