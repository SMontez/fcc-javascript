Original Code

const createPerson = (name, age, gender) => {
"use strict";
return {
name: name,
age: age,
gender: gender
};
};
console.log(createPerson("Zodiac Hasbro", 56, "male")); // returns a proper object

New Code

const createPerson = (name, age, gender) => {
"use strict";
return {name, age, gender};
};
console.log(createPerson("Zodiac Hasbro", 56, "male")); // returns a proper object
