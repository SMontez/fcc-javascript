• Home > Exploring ES6 Features - Mutate an Array Declared with const

Original Code

const s = [5, 7, 2];
function editInPlace() {
"use strict";
// s = [2, 5, 7]; <- this is invalid
}
editInPlace();

New Code

const s = [5, 7, 2];
function editInPlace() {
"use strict";
s[0] = 2;
s[1] = 5;
s[2] = 7;
}
editInPlace();
